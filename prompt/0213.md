下面是一套**可直接跑的 Python 3.12 模板**：从 `.xlsx` 里读取列 `xunwenneirong`，对长笔录**分块（带 overlap）抽取**，每块让模型输出**严格 JSON**，然后**合并去重/冲突标记**，最后把结果**回写到同一个 xlsx**（新增列）。

我按你场景把**“字段—标签表”嵌进提示词**里了，并且做了：

* **OpenAI 兼容接口（llama.cpp --server）优先**
* 可选 **命令行调用**（llama.cpp main / llama-cli）
* JSON 解析失败自动重试
* 合并：同字段多值 → `values[] + resolved + conflict_flag + evidence_quotes[]`

# =========================
# 工具函数：分块 / JSON修复 / 合并
# =========================
def chunk_text(text: str, chunk_chars: int, overlap_chars: int, max_chunks: int) -> List[Tuple[int, int, str]]:
    text = (text or "").strip()
    if not text:
        return []

    chunks = []
    start = 0
    n = len(text)
    idx = 0
    while start < n and idx < max_chunks:
        end = min(n, start + chunk_chars)
        chunk = text[start:end]
        chunks.append((start, end, chunk))
        idx += 1
        if end >= n:
            break
        start = max(0, end - overlap_chars)
    return chunks


def extract_json_str(s: str) -> str:
    """
    尽量从输出里抠出JSON对象字符串（防模型偶尔带前后文本）
    """
    s = s.strip()
    # 直接就是json
    if s.startswith("{") and s.endswith("}"):
        return s
    # 找第一个{ 到最后一个}
    i = s.find("{")
    j = s.rfind("}")
    if i != -1 and j != -1 and j > i:
        return s[i:j+1]
    return s


def safe_json_loads(s: str) -> Optional[Dict[str, Any]]:
    try:
        return json.loads(s)
    except Exception:
        return None


def stable_hash(text: str) -> str:
    return hashlib.md5(text.encode("utf-8", errors="ignore")).hexdigest()


@dataclass
class LLMResult:
    ok: bool
    data: Optional[Dict[str, Any]] = None
    raw: str = ""
    error: str = ""


# =========================
# llama.cpp 调用：OpenAI server
# =========================
def call_llama_openai_server(system_prompt: str, user_prompt: str) -> str:
    url = f"{OPENAI_BASE_URL}/chat/completions"
    payload = {
        "model": OPENAI_MODEL_NAME,
        "temperature": TEMPERATURE,
        "top_p": TOP_P,
        "max_tokens": MAX_TOKENS,
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ],
    }
    r = requests.post(url, json=payload, timeout=REQUEST_TIMEOUT)
    r.raise_for_status()
    obj = r.json()
    return obj["choices"][0]["message"]["content"]


# =========================
# llama.cpp 调用：命令行（备用）
# =========================
def call_llama_cli(full_prompt: str) -> str:
    """
    需要你的 llama-cli.exe / main.exe 支持读取prompt并输出。
    这里给模板：你可能要按自己编译版本的参数微调。
    """
    import subprocess

    cmd = [
        LLAMA_CLI_EXE,
        "-m", LLAMA_MODEL_PATH,
        "-c", str(CLI_CTX),
        "--temp", str(TEMPERATURE),
        "--top-p", str(TOP_P),
        "-p", full_prompt,
        "-n", str(MAX_TOKENS),
    ]
    # Windows 下建议设置 encoding
    p = subprocess.run(cmd, capture_output=True, text=True, encoding="utf-8", errors="ignore")
    if p.returncode != 0:
        raise RuntimeError(p.stderr or "llama-cli failed")
    return p.stdout.strip()


# =========================
# 单块抽取
# =========================
def extract_from_chunk(chunk_text: str, chunk_index: int, cstart: int, cend: int) -> LLMResult:
    system_prompt = FIELD_SPEC.strip()

    user_prompt = f"""
下面是讯问内容的一个分块（chunk）。请严格按规范输出JSON。

chunk_index: {chunk_index}
chunk_char_start: {cstart}
chunk_char_end: {cend}

讯问内容chunk：
{chunk_text}
""".strip()

    last_err = ""
    for attempt in range(MAX_RETRY + 1):
        try:
            if USE_OPENAI_SERVER:
                raw = call_llama_openai_server(system_prompt, user_prompt)
            elif USE_CLI:
                full_prompt = system_prompt + "\n\n" + user_prompt
                raw = call_llama_cli(full_prompt)
            else:
                raise RuntimeError("No LLM backend configured (USE_OPENAI_SERVER/USE_CLI).")

            js = extract_json_str(raw)
            data = safe_json_loads(js)
            if data is None:
                last_err = f"JSON parse failed. raw={raw[:200]}"
                # 轻量“修复式重试”：让模型只输出JSON
                user_prompt = (
                    user_prompt
                    + "\n\n你的上次输出不是严格JSON。请只输出一个JSON对象，不要任何解释。"
                )
                time.sleep(SLEEP_BETWEEN_RETRY)
                continue

            # 补齐 doc_meta（以防模型漏）
            data.setdefault("doc_meta", {})
            data["doc_meta"].update({
                "chunk_index": chunk_index,
                "chunk_char_start": cstart,
                "chunk_char_end": cend,
            })
            return LLMResult(ok=True, data=data, raw=raw)
        except Exception as e:
            last_err = str(e)
            time.sleep(SLEEP_BETWEEN_RETRY)

    return LLMResult(ok=False, data=None, raw="", error=last_err)


# =========================
# 合并策略：values去重 + resolved + 冲突标记 + 证据聚合
# =========================
def merge_chunk_results(chunk_jsons: List[Dict[str, Any]]) -> Dict[str, Any]:
    merged: Dict[str, Any] = {
        "merged_fields": {},
        "conflict_flag": False,
        "evidence_quotes": {},
        "chunk_count": len(chunk_jsons),
    }

    # 统一字段名集合
    field_names = set()
    for cj in chunk_jsons:
        fields = (cj or {}).get("fields", {}) or {}
        field_names.update(fields.keys())

    def add_quote(fname: str, quote: str):
        if not quote:
            return
        merged["evidence_quotes"].setdefault(fname, [])
        # 去重
        if quote not in merged["evidence_quotes"][fname]:
            merged["evidence_quotes"][fname].append(quote)

    for fname in sorted(field_names):
        vals: List[str] = []
        quotes: List[str] = []

        for cj in chunk_jsons:
            fields = (cj or {}).get("fields", {}) or {}
            fobj = fields.get(fname)
            if not fobj:
                continue

            # 兼容 label / values 两种
            if isinstance(fobj, dict):
                if "values" in fobj and isinstance(fobj["values"], list):
                    for v in fobj["values"]:
                        if isinstance(v, str) and v and v != "不详":
                            vals.append(v)
                if "label" in fobj and isinstance(fobj["label"], str):
                    v = fobj["label"].strip()
                    if v and v != "不详":
                        vals.append(v)
                q = fobj.get("quote", "")
                if isinstance(q, str) and q.strip():
                    quotes.append(q.strip())
                    add_quote(fname, q.strip())

        # 去重保持顺序
        dedup_vals = []
        for v in vals:
            v = v.strip()
            if v and v not in dedup_vals:
                dedup_vals.append(v)

        conflict = len(dedup_vals) > 1
        if conflict:
            merged["conflict_flag"] = True

        # resolved：简单规则——若有多个取第一个（更稳：你可以后续加“末份优先/补充优先”）
        resolved = dedup_vals[0] if dedup_vals else "不详"

        merged["merged_fields"][fname] = {
            "values": dedup_vals if dedup_vals else [],
            "resolved": resolved,
            "conflict": conflict,
        }

    return merged


# =========================
# Excel 读写工具
# =========================
def find_col_index_by_header(ws, header_name: str) -> Optional[int]:
    header_row = 1
    for col in range(1, ws.max_column + 1):
        v = ws.cell(row=header_row, column=col).value
        if isinstance(v, str) and v.strip() == header_name:
            return col
    return None


def ensure_col(ws, header_name: str) -> int:
    col = find_col_index_by_header(ws, header_name)
    if col is not None:
        return col
    # append new column
    new_col = ws.max_column + 1
    ws.cell(row=1, column=new_col).value = header_name
    return new_col


def cell_str(v: Any) -> str:
    if v is None:
        return ""
    if isinstance(v, str):
        return v
    return str(v)


# =========================
# 主流程
# =========================
def process_workbook(path: str):
    wb = load_workbook(path)
    ws = wb[SHEET_NAME] if SHEET_NAME else wb.active

    text_col = find_col_index_by_header(ws, TEXT_COL_NAME)
    if text_col is None:
        raise ValueError(f"找不到列名: {TEXT_COL_NAME}")

    out_col_all = ensure_col(ws, OUT_COL_JSON_ALL)
    out_col_merged = ensure_col(ws, OUT_COL_JSON_MERGED)
    out_col_summary = ensure_col(ws, OUT_COL_SUMMARY)
    out_col_conflict = ensure_col(ws, OUT_COL_CONFLICT)

    # 可选：缓存，避免重复文本重复跑
    cache: Dict[str, Dict[str, Any]] = {}

    total_rows = ws.max_row - 1
    done = 0

    for r in range(2, ws.max_row + 1):
        text = cell_str(ws.cell(row=r, column=text_col).value).strip()
        if not text:
            continue

        h = stable_hash(text)
        if h in cache:
            result_pack = cache[h]
        else:
            chunks = chunk_text(text, CHUNK_CHARS, OVERLAP_CHARS, MAX_CHUNKS)
            chunk_results = []
            for idx, (cstart, cend, ctext) in enumerate(chunks):
                res = extract_from_chunk(ctext, idx, cstart, cend)
                if not res.ok:
                    # 失败也记录，别直接丢
                    chunk_results.append({
                        "doc_meta": {"chunk_index": idx, "chunk_char_start": cstart, "chunk_char_end": cend},
                        "fields": {},
                        "error": res.error,
                        "raw_head": res.raw[:200] if res.raw else "",
                    })
                else:
                    chunk_results.append(res.data)

            merged = merge_chunk_results(chunk_results)

            # 给人工扫一眼的summary（非常简短）
            mf = merged["merged_fields"]
            def rv(k: str) -> str:
                return mf.get(k, {}).get("resolved", "不详")

            summary = (
                f"关系:{rv('relation')} 渠道:{rv('meet_channel')} "
                f"地点:{rv('place_type')} 时段:{rv('time_range')} "
                f"留宿:{rv('overnight')} 开房:{rv('open_room')} "
                f"诱骗:{rv('lure_main')} 偷拍:{rv('record_video')} "
                f"威胁:{rv('threaten')} 报案延迟:{rv('report_delay_range')}"
            )

            result_pack = {
                "chunk_results": chunk_results,
                "merged": merged,
                "summary": summary,
            }
            cache[h] = result_pack

        # 写回Excel
        ws.cell(row=r, column=out_col_all).value = json.dumps(result_pack["chunk_results"], ensure_ascii=False)
        ws.cell(row=r, column=out_col_merged).value = json.dumps(result_pack["merged"], ensure_ascii=False)
        ws.cell(row=r, column=out_col_summary).value = result_pack["summary"]
        ws.cell(row=r, column=out_col_conflict).value = "是" if result_pack["merged"].get("conflict_flag") else "否"

        done += 1
        if done % 5 == 0:
            print(f"[{done}/{total_rows}] processed...")

    # 保存为新文件，防止覆盖
    out_path = re.sub(r"\.xlsx$", "", path, flags=re.I) + "_with_llm.xlsx"
    wb.save(out_path)
    print(f"完成，已输出：{out_path}")


if __name__ == "__main__":
    process_workbook(INPUT_XLSX)

def stable_hash(text: str) -> str:
    return hashlib.md5(text.encode("utf-8", errors="ignore")).hexdigest()


def extract_json_str(s: str) -> str:
    s = (s or "").strip()
    if s.startswith("{") and s.endswith("}"):
        return s
    i = s.find("{")
    j = s.rfind("}")
    if i != -1 and j != -1 and j > i:
        return s[i:j+1]
    return s


def safe_json_loads(s: str) -> Optional[Dict[str, Any]]:
    try:
        return json.loads(s)
    except Exception:
        return None


def chunk_text(text: str, chunk_chars: int, overlap_chars: int, max_chunks: int) -> List[Tuple[int, int, str]]:
    text = (text or "").strip()
    if not text:
        return []
    chunks = []
    start = 0
    n = len(text)
    idx = 0
    while start < n and idx < max_chunks:
        end = min(n, start + chunk_chars)
        chunks.append((start, end, text[start:end]))
        idx += 1
        if end >= n:
            break
        start = max(0, end - overlap_chars)
    return chunks


def parse_dt(v: Any) -> Optional[datetime]:
    if v is None:
        return None
    if isinstance(v, datetime):
        return v
    s = str(v).strip()
    if not s:
        return None
    try:
        # 支持 "2026-02-12 10:20:00" / "2026/2/12 10:20" / excel序列等一般情况
        return dtparser.parse(s, fuzzy=True)
    except Exception:
        return None


def is_valid_cn_id(idno: str) -> bool:
    if not idno:
        return False
    idno = idno.strip()
    return bool(re.fullmatch(r"\d{15}|\d{17}[\dXx]", idno))


def birthdate_from_cn_id(idno: str) -> Optional[date]:
    if not is_valid_cn_id(idno):
        return None
    idno = idno.strip()
    try:
        if len(idno) == 18:
            b = idno[6:14]  # YYYYMMDD
            y, m, d = int(b[0:4]), int(b[4:6]), int(b[6:8])
            return date(y, m, d)
        else:  # 15位：YYMMDD（按19YY处理，够用；极少见跨2000的15位）
            b = idno[6:12]  # YYMMDD
            y, m, d = int("19" + b[0:2]), int(b[2:4]), int(b[4:6])
            return date(y, m, d)
    except Exception:
        return None


def calc_age_on(birth: date, on_date: date) -> int:
    if not birth or not on_date:
        return 0
    years = on_date.year - birth.year
    if (on_date.month, on_date.day) < (birth.month, birth.day):
        years -= 1
    return max(0, years)

def call_llama_openai_server(system_prompt: str, user_prompt: str) -> str:
    url = f"{OPENAI_BASE_URL}/chat/completions"
    payload = {
        "model": OPENAI_MODEL_NAME,
        "temperature": TEMPERATURE,
        "top_p": TOP_P,
        "max_tokens": MAX_TOKENS,
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ],
    }
    r = requests.post(url, json=payload, timeout=REQUEST_TIMEOUT)
    r.raise_for_status()
    obj = r.json()
    return obj["choices"][0]["message"]["content"]


@dataclass
class LLMResult:
    ok: bool
    data: Optional[Dict[str, Any]] = None
    raw: str = ""
    error: str = ""


def extract_from_chunk(ctext: str, chunk_index: int, cstart: int, cend: int) -> LLMResult:
    system_prompt = FIELD_SPEC.strip()
    user_prompt = f"""
下面是讯问内容的一个分块（chunk）。请严格按规范输出JSON。

chunk_index: {chunk_index}
chunk_char_start: {cstart}
chunk_char_end: {cend}

讯问内容chunk：
{ctext}
""".strip()

    last_err = ""
    for attempt in range(MAX_RETRY + 1):
        try:
            raw = call_llama_openai_server(system_prompt, user_prompt)
            js = extract_json_str(raw)
            data = safe_json_loads(js)
            if data is None:
                last_err = f"JSON parse failed"
                user_prompt += "\n\n你的上次输出不是严格JSON。请只输出一个JSON对象，不要任何解释。"
                time.sleep(SLEEP_BETWEEN_RETRY)
                continue

            data.setdefault("doc_meta", {})
            data["doc_meta"].update({
                "chunk_index": chunk_index,
                "chunk_char_start": cstart,
                "chunk_char_end": cend,
            })
            return LLMResult(ok=True, data=data, raw=raw)
        except Exception as e:
            last_err = str(e)
            time.sleep(SLEEP_BETWEEN_RETRY)

    return LLMResult(ok=False, error=last_err)

def merge_chunk_results(chunk_jsons: List[Dict[str, Any]]) -> Dict[str, Any]:
    """
    chunk级合并：按出现顺序保留values并去重，同时生成 resolved（首优先/末优先）
    注意：这里的“首/末”是 chunk 顺序；由于你一行就是一份笔录，这通常够用。
    """
    merged: Dict[str, Any] = {
        "merged_fields": {},
        "conflict_flag": False,
        "evidence_quotes": {},
        "chunk_count": len(chunk_jsons),
    }

    # 统一字段名集合
    field_names = set()
    for cj in chunk_jsons:
        fields = (cj or {}).get("fields", {}) or {}
        field_names.update(fields.keys())

    def add_quote(fname: str, quote: str):
        if not quote:
            return
        merged["evidence_quotes"].setdefault(fname, [])
        if quote not in merged["evidence_quotes"][fname]:
            merged["evidence_quotes"][fname].append(quote)

    def collect_values(fname: str) -> Tuple[List[str], List[str]]:
        vals: List[str] = []
        quotes: List[str] = []
        for cj in chunk_jsons:
            fobj = ((cj or {}).get("fields", {}) or {}).get(fname)
            if not isinstance(fobj, dict):
                continue
            # values
            if "values" in fobj and isinstance(fobj["values"], list):
                for v in fobj["values"]:
                    if isinstance(v, str) and v.strip() and v.strip() != "不详":
                        vals.append(v.strip())
            # label
            if "label" in fobj and isinstance(fobj["label"], str):
                v = fobj["label"].strip()
                if v and v != "不详":
                    vals.append(v)
            # quote
            q = fobj.get("quote", "")
            if isinstance(q, str) and q.strip():
                q = q.strip()
                quotes.append(q)
                add_quote(fname, q)
        # 去重保序
        dedup = []
        for v in vals:
            if v not in dedup:
                dedup.append(v)
        return dedup, quotes

    for fname in sorted(field_names):
        dedup_vals, _quotes = collect_values(fname)
        conflict = len(dedup_vals) > 1
        if conflict:
            merged["conflict_flag"] = True

        first_val = dedup_vals[0] if dedup_vals else "不详"
        last_val = dedup_vals[-1] if dedup_vals else "不详"

        if fname in FIRST_PRIORITY_FIELDS:
            resolved = first_val
            resolve_rule = "first"
        elif fname in LAST_PRIORITY_FIELDS:
            resolved = last_val
            resolve_rule = "last"
        else:
            # 默认：若存在多个值，先取 last（更靠近证据定型），否则取 first
            resolved = last_val if conflict else first_val
            resolve_rule = "last_if_conflict_else_first"

        merged["merged_fields"][fname] = {
            "values": dedup_vals,
            "first": first_val,
            "last": last_val,
            "resolved": resolved,
            "conflict": conflict,
            "resolve_rule": resolve_rule
        }

    return merged

def find_col_index_by_header(ws, header_name: str) -> Optional[int]:
    for col in range(1, ws.max_column + 1):
        v = ws.cell(row=1, column=col).value
        if isinstance(v, str) and v.strip() == header_name:
            return col
    return None

    if col is not None:
        return col
    new_col = ws.max_column + 1
    ws.cell(row=1, column=new_col).value = header_name
    return new_col

def normalize_role(role: str) -> str:

    mapping = {

def pick_case_level_value(records_sorted: List[Dict[str, Any]], field_name: str,
                          prefer: str = "first") -> str:
    """
    records_sorted: 已按笔录次数/时间升序的笔录记录列表
    prefer: "first" 或 "last" 或 "any_yes"（是/否字段取或）
    依赖每条记录里已存在 merged_fields[field]['resolved'/'first'/'last'] 结构
    """
    vals = []
    for rec in records_sorted:
        mf = (rec.get("llm_merged") or {}).get("merged_fields") or {}
        f = mf.get(field_name) or {}
        v = f.get("resolved", "不详")
        if v and v != "不详":
            vals.append(v)

    if prefer == "first":
        return vals[0] if vals else "不详"
    if prefer == "last":
        return vals[-1] if vals else "不详"
    if prefer == "any_yes":
        # 适用于 是/否/不详
        for v in vals:
            if v == "是":
                return "是"
        # 如果没“是”，有“否”则否，否则不详
        for v in vals:
            if v == "否":
                return "否"
        return "不详"
    return vals[-1] if vals else "不详"


def aggregate_case_sheet(rows: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """
    输入：笔录级行记录（含案件编号、人员类型、笔录次数、开始时间、llm_merged）
    输出：案件级汇总行列表（每案一行）
    """
    # 按案件编号分组
    by_case: Dict[str, List[Dict[str, Any]]] = {}
    for r in rows:
        cid = (r.get("case_id") or "").strip()
        if not cid:
            continue
        by_case.setdefault(cid, []).append(r)

    out_rows: List[Dict[str, Any]] = []

    for cid, recs in by_case.items():
        # 先按笔录次数/开始时间排序（升序）
        def sort_key(x):
            return (x.get("times", 0), x.get("start_dt") or datetime.min)
        recs_sorted = sorted(recs, key=sort_key)

        # 再按人员类型分组（每个角色可多份笔录）
        by_role: Dict[str, List[Dict[str, Any]]] = {}
        for r in recs_sorted:
            role = normalize_role(r.get("role", "不详"))
            by_role.setdefault(role, []).append(r)

        # 案件级：你最常用的“规律字段”
        # 关系/渠道/地点/诱骗：优先受害人（更贴近真实经历）；若无受害人，用嫌疑人补
        victim_recs = by_role.get("受害人", [])
        suspect_recs = by_role.get("嫌疑人", [])
        reporter_recs = by_role.get("报案人", [])

        def pick_from_roles(field, prefer_first_role=True, prefer="first"):
            if victim_recs:
                return pick_case_level_value(victim_recs, field, prefer=prefer)
            if suspect_recs:
                return pick_case_level_value(suspect_recs, field, prefer=prefer)
            # 兜底：全案
            return pick_case_level_value(recs_sorted, field, prefer=prefer)

        # 是否类字段：任一为“是”则是
        def pick_any_yes(field):
            return pick_case_level_value(recs_sorted, field, prefer="any_yes")

        case_row = {
            "案件编号": cid,
            "笔录总数": len(recs_sorted),

            # 核心规律字段（案件级）
            "关系": pick_from_roles("relation", prefer="first"),
            "认识渠道": pick_from_roles("meet_channel", prefer="first"),
            "约见方式": pick_from_roles("invite_channel", prefer="first"),
            "案发地点类型": pick_from_roles("place_type", prefer="first"),
            "案发时间段": pick_from_roles("time_range", prefer="first"),
            "案发日期类型": pick_from_roles("date_type", prefer="first"),
            "诱骗方式(主要)": pick_from_roles("lure_main", prefer="first"),

            "是否留宿": pick_any_yes("overnight"),
            "是否开房": pick_any_yes("open_room"),
            "是否饮酒": pick_any_yes("alcohol"),
            "是否用药": pick_any_yes("drugged"),
            "是否威胁": pick_any_yes("threaten"),
            "是否偷拍视频": pick_any_yes("record_video"),
            "是否传播/勒索": pick_any_yes("spread_or_extort"),

            # 定罪/补证更强字段：末份优先（嫌疑人优先）
            "是否明知未成年": (pick_case_level_value(suspect_recs, "suspect_knew_minor", prefer="last")
                          if suspect_recs else pick_case_level_value(recs_sorted, "suspect_knew_minor", prefer="last")),

            "是否多次发生": pick_from_roles("repeat", prefer="last"),
            "发生次数区间": pick_from_roles("repeat_count_range", prefer="last"),

            # 报案相关：优先报案人最早信息
            "报案渠道": (pick_case_level_value(reporter_recs, "report_channel", prefer="first")
                      if reporter_recs else pick_case_level_value(recs_sorted, "report_channel", prefer="first")),
            "报案延迟区间": (pick_case_level_value(reporter_recs, "report_delay_range", prefer="first")
                        if reporter_recs else pick_case_level_value(recs_sorted, "report_delay_range", prefer="first")),
            "是否就医": pick_any_yes("medical_visit"),

            # 冲突标记：任一笔录chunk合并存在冲突
            "冲突标记": ("是" if any((r.get("llm_merged") or {}).get("conflict_flag") for r in recs_sorted) else "否"),
        }

        out_rows.append(case_row)

    return out_rows

# 主流程：笔录级抽取 + 回写 + 生成案件级sheet
# =========================
def process_workbook(path: str):
    wb = load_workbook(path)
    ws = wb[SHEET_NAME] if SHEET_NAME else wb.active

    # 必要列
    col_case = find_col_index_by_header(ws, COL_CASE_ID)
    col_text = find_col_index_by_header(ws, COL_TRANSCRIPT)
    col_role = find_col_index_by_header(ws, COL_ROLENAME)
    col_times = find_col_index_by_header(ws, COL_TIMES)
    col_start = find_col_index_by_header(ws, COL_START_TIME)

    if col_case is None or col_text is None or col_role is None or col_times is None or col_start is None:
        raise ValueError("缺少必要列：案件编号/xunwenneirong/人员类型/笔录次数/开始时间，请检查列名。")

    # 可选列
    col_idno = find_col_index_by_header(ws, COL_IDNO)
    col_age = find_col_index_by_header(ws, COL_AGE)

    # 输出列（笔录级）
    out_all = ensure_col(ws, OUT_COL_JSON_ALL)
    out_merged = ensure_col(ws, OUT_COL_JSON_MERGED)
    out_sum = ensure_col(ws, OUT_COL_SUMMARY)
    out_conf = ensure_col(ws, OUT_COL_CONFLICT)

    # 若需要写回修复后的年龄
    out_age_fixed = None
    if col_age is not None and col_idno is not None:
        out_age_fixed = ensure_col(ws, "年龄_修复")

    cache: Dict[str, Dict[str, Any]] = {}
    transcript_rows_for_agg: List[Dict[str, Any]] = []

    total_rows = ws.max_row - 1
    done = 0

    for r in range(2, ws.max_row + 1):
        cid = cell_str(ws.cell(row=r, column=col_case).value).strip()
        role = normalize_role(cell_str(ws.cell(row=r, column=col_role).value))
        times = cell_int(ws.cell(row=r, column=col_times).value)
        start_dt = parse_dt(ws.cell(row=r, column=col_start).value)

        text = cell_str(ws.cell(row=r, column=col_text).value).strip()
        if not text:
            continue

        # 年龄修复（仅当原年龄为0或空）
        age_fixed = None
        if out_age_fixed is not None:
            age_val = cell_int(ws.cell(row=r, column=col_age).value)
            idno = cell_str(ws.cell(row=r, column=col_idno).value).strip()
            if (age_val == 0 or age_val < 0) and is_valid_cn_id(idno) and start_dt:
                bd = birthdate_from_cn_id(idno)
                if bd:
                    age_fixed = calc_age_on(bd, start_dt.date())
                    ws.cell(row=r, column=out_age_fixed).value = age_fixed
            else:
                ws.cell(row=r, column=out_age_fixed).value = age_val if age_val else ""

        # LLM抽取（缓存去重）
        h = stable_hash(text)
        if h in cache:
            pack = cache[h]
        else:
            chunks = chunk_text(text, CHUNK_CHARS, OVERLAP_CHARS, MAX_CHUNKS)
            chunk_results = []
            for idx, (cstart, cend, ctext) in enumerate(chunks):
                res = extract_from_chunk(ctext, idx, cstart, cend)
                if not res.ok:
                    chunk_results.append({
                        "doc_meta": {"chunk_index": idx, "chunk_char_start": cstart, "chunk_char_end": cend},
                        "fields": {},
                        "error": res.error,
                    })
                else:
                    chunk_results.append(res.data)

            merged = merge_chunk_results(chunk_results)

            mf = merged["merged_fields"]
            def rv(k: str) -> str:
                return (mf.get(k, {}) or {}).get("resolved", "不详")

            summary = (
                f"关系:{rv('relation')} 渠道:{rv('meet_channel')} 地点:{rv('place_type')} "
                f"时段:{rv('time_range')} 留宿:{rv('overnight')} 开房:{rv('open_room')} "
                f"诱骗:{rv('lure_main')} 偷拍:{rv('record_video')} 威胁:{rv('threaten')} "
                f"明知未成年:{rv('suspect_knew_minor')} 报案延迟:{rv('report_delay_range')}"
            )

            pack = {"chunk_results": chunk_results, "merged": merged, "summary": summary}
            cache[h] = pack

        # 回写笔录级
        ws.cell(row=r, column=out_all).value = json.dumps(pack["chunk_results"], ensure_ascii=False)
        ws.cell(row=r, column=out_merged).value = json.dumps(pack["merged"], ensure_ascii=False)
        ws.cell(row=r, column=out_sum).value = pack["summary"]
        ws.cell(row=r, column=out_conf).value = "是" if pack["merged"].get("conflict_flag") else "否"

        # 为案件级聚合准备结构化行
        transcript_rows_for_agg.append({
            "case_id": cid,
            "role": role,
            "times": times,
            "start_dt": start_dt,
            "llm_merged": pack["merged"]
        })

        done += 1
        if done % 5 == 0:
            print(f"[{done}/{total_rows}] processed...")

    # 生成案件级sheet
    case_rows = aggregate_case_sheet(transcript_rows_for_agg)
    if "案件级汇总" in wb.sheetnames:
        del wb["案件级汇总"]
    ws_case = wb.create_sheet("案件级汇总")

    # 写表头
    if case_rows:
        headers = list(case_rows[0].keys())
        for c, h in enumerate(headers, start=1):
            ws_case.cell(row=1, column=c).value = h
        # 写数据
        for i, row in enumerate(case_rows, start=2):
            for c, h in enumerate(headers, start=1):
                ws_case.cell(row=i, column=c).value = row.get(h, "")

    out_path = re.sub(r"\.xlsx$", "", path, flags=re.I) + "_with_llm.xlsx"
    wb.save(out_path)
    print(f"完成，已输出：{out_path}")


if __name__ == "__main__":
    process_workbook(INPUT_XLSX)

## 关于你说的“首份更详细”我已经体现在规则里了

* 过程画像类字段（关系/渠道/地点/诱骗/带离等）：`首份优先`
* 补证定罪类字段（明知未成年/偷拍视频/威胁/证据类型等）：`末份优先`
* 并保留 `values / first / last / resolved / conflict` 便于后续你做“首供 vs 末供变化规律”。
