基于你的内网无互联网环境和“标准件”任务下发的工作模式，**生成 Excel (.xlsx) 是最合适的方案**。

**原因分析：**
1.  **易于分发**：Excel文件可以通过内网邮件或OA系统直接作为附件发送给派出所。
2.  **便于反馈**：派出所民警可以在Excel中直接增加一列“核查情况”进行填写，反馈后再汇总，符合“闭环”逻辑。
3.  **无需额外环境**：内网电脑通常都装有Office/WPS，不需要浏览器兼容性调试，也不需要复杂的Web服务。
===
这个修改非常符合公安实战中**“一案一档、一人一档”**的数据处理逻辑！

将数据“展开”（行转列/爆炸裂变）后，每个成员占据独立的一行，且身份证号码独立成列。这样做有两大核心优势：
1. **便于批量比对**：你可以直接复制一整列“团伙成员身份证号码”，丢进公安大数据的“重口/前科人员比对模型”中进行批量碰撞。
2. **逻辑更清晰**：“序号”列完美地将他们捆绑在同一个“聚集事件/团伙”中，基层民警核查时可以按序号“一锅端”。

以下是为你**重构后**的 Python 脚本。你可以直接用它替换掉之前的代码。

### 优化后的 Python 脚本

```python
import pandas as pd
import re
from datetime import datetime

# =================配置区域=================
INPUT_FILE = 'source_data.csv'  # 你的原始数据文件名
OUTPUT_FILE = f'未成年聚集研判_指令下发单_{datetime.now().strftime("%Y%m%d")}.xlsx'
CURRENT_YEAR = 2026 # 用于计算年龄

# =================核心逻辑=================

def parse_complex_text(text):
    """
    解析复杂的文本字段，提取时间、地点、坐标和人员明细
    """
    if not isinstance(text, str):
        return None
    
    # 提取时间、地点、经纬度
    base_pattern = re.search(r'于(.*?)期间，在(.*?)\n?（(.*?)）有聚集', text, re.DOTALL)
    result = {
        '聚集时间段': '',
        '聚集地点': '',
        '经度': '',
        '纬度': '',
        '聚集人数': 0,
        '人员详情': []
    }
    
    if base_pattern:
        result['聚集时间段'] = base_pattern.group(1).strip()
        result['聚集地点'] = base_pattern.group(2).strip().replace('\n', '')
        coords = base_pattern.group(3).strip().split('、')
        if len(coords) >= 2:
            result['经度'] = coords[0]
            result['纬度'] = coords[1]
    
    # 提取人数
    count_match = re.search(r'聚集号码数量：(\d+)个', text)
    if count_match:
        result['聚集人数'] = int(count_match.group(1))

    # 提取人员名单
    if "聚集学生为：" in text:
        people_text = text.split("聚集学生为：")[1]
        people_items = re.findall(r'(\d{11})（(\d{18}|.*?)、(.*?)）', people_text)
        
        for p in people_items:
            phone = p[0]
            id_card = p[1]
            name = p[2]
            
            # 算年龄
            age = 0
            if len(id_card) == 18:
                try:
                    birth_year = int(id_card[6:10])
                    age = CURRENT_YEAR - birth_year
                except ValueError:
                    age = 0
            
            result['人员详情'].append({
                '姓名': name,
                '身份证': id_card,
                '电话': phone,
                '年龄': age
            })
            
    return result

def determine_risk_level(location, time_range, count):
    """判断风险标签"""
    risk_tags = []
    location = str(location)
    time_range = str(time_range)
    
    high_risk_locs = ['楼顶', '天台', '河边', '水库', '网吧', '酒吧', '废弃', 'KTV', '桥底']
    for loc in high_risk_locs:
        if loc in location:
            risk_tags.append(f'高危地点({loc})')
            break
            
    if re.search(r'\s0[0-4]:', time_range):
        risk_tags.append('凌晨聚集')
        
    if count >= 5:
        risk_tags.append('多人团伙')
        
    return " | ".join(risk_tags) if risk_tags else "一般关注"

def main():
    print(f"正在读取文件: {INPUT_FILE} ...")
    try:
        try:
            df = pd.read_csv(INPUT_FILE, encoding='utf-8')
        except UnicodeDecodeError:
            df = pd.read_csv(INPUT_FILE, encoding='gbk')
    except Exception as e:
        print(f"读取失败，请检查文件是否存在。错误: {e}")
        return

    # 自动寻找包含数据的目标列
    target_col = None
    for col in df.columns:
        first_valid = df[col].dropna().astype(str).iloc[0] if not df[col].dropna().empty else ""
        if "聚集学生为" in first_valid:
            target_col = col
            break
            
    if not target_col:
        print("未找到包含聚集信息的列，请检查CSV表头。")
        return

    print("正在解析并重构数据结构（行转列处理）...")
    
    # 第一步：解析并按事件去重（字典形式保存，Key为地点+时间）
    unique_events = {}
    for index, row in df.iterrows():
        raw_text = row[target_col]
        parsed = parse_complex_text(raw_text)
        
        if parsed and parsed['聚集地点']:
            event_id = f"{parsed['聚集地点']}_{parsed['聚集时间段']}"
            if event_id not in unique_events:
                unique_events[event_id] = parsed

    # 第二步：将每个事件里的每个成员“展开”成独立的一行
    exploded_rows = []
    group_serial_number = 1  # 团伙序号计数器
    
    for event_id, parsed_data in unique_events.items():
        members = parsed_data['人员详情']
        if not members:
            continue
            
        # 找出该团伙的“带头人”（年龄最大的）
        leader = max(members, key=lambda x: x['年龄'])
        leader_name_age = f"{leader['姓名']}({leader['年龄']}岁)"
        leader_id_card = leader['身份证']
        
        # 计算风险标签和管控建议
        risk_label = determine_risk_level(parsed_data['聚集地点'], parsed_data['聚集时间段'], parsed_data['聚集人数'])
        advice = '建议约谈监护人，核查聚集事由'
        if '楼顶' in parsed_data['聚集地点']:
            advice = '【防坠亡/防极端】立即核查天台门锁，清查现场是否遗留违禁品/烟酒，核实是否存在欺凌。'
            
        # 遍历该团伙的每一个成员，生成一行数据
        for member in members:
            row_data = {
                '序号': group_serial_number,
                '风险标签': risk_label,
                '聚集地点': parsed_data['聚集地点'],
                '聚集时间段': parsed_data['聚集时间段'],
                '聚集人数': parsed_data['聚集人数'],
                '团伙成员': f"{member['姓名']}({member['年龄']}岁)",
                '团伙成员身份证号码': member['身份证'],
                '团伙成员手机号': member['电话'], # 顺手加了个手机号，方便派出所联系
                '带头/年长者': leader_name_age,
                '带头/年长者身份证号码': leader_id_card,
                '管控建议': advice
            }
            exploded_rows.append(row_data)
            
        # 团伙序号+1
        group_serial_number += 1

    # 转换为DataFrame
    result_df = pd.DataFrame(exploded_rows)
    if result_df.empty:
        print("没有解析到有效数据。")
        return

    # =================输出Excel=================
    print(f"正在生成标准派单Excel: {OUTPUT_FILE} ...")
    
    # 按照要求的列顺序排列
    output_cols = [
        '序号', '风险标签', '聚集地点', '聚集时间段', '聚集人数', 
        '团伙成员', '团伙成员身份证号码', '团伙成员手机号',
        '带头/年长者', '带头/年长者身份证号码', '管控建议'
    ]
    
    with pd.ExcelWriter(OUTPUT_FILE, engine='openpyxl') as writer:
        # 下发单
        result_df[output_cols].to_excel(writer, sheet_name='标准化指令下发单', index=False)
        
    print("处理完成！数据已按团伙人员成功拆分为多行。")

if __name__ == '__main__':
    main()
```

### 基层收到表后怎么用？（建议在发“标准件”时附带的说明）

发任务给派出所时，可以让他们这么操作：
1. **看序号**：相同序号的是同一个案子/线索，派给同一个警组去跟进。
2. **看带头人**：直接把“带头/年长者”传唤到派出所了解情况，擒贼先擒王。
3. **查身份**：直接复制“团伙成员身份证号码”那一整列，去公安部/省厅的大数据平台上执行**“批量人员研判”**（查吸毒、查前科、查在逃、查是否为涉诈“两卡”开卡人），这种单行单列的格式是最兼容公安系统批量查询功能的。
