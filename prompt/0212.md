        ```sql
            SELECT
                jq."calltime" AS "报警时间",
                jq."caseno" AS "警情编号",
                jq."dutydeptname" AS "管辖单位",
                jq."cmdname" AS "分局",

                zza."ajxx_ajbh" "案件编号",
                zza."ajxx_lasj" 立案时间,
                zza."ajxx_cbdw_mc" 办案单位
            FROM ywdata."zq_kshddpt_dsjfx_jq" jq
            LEFT JOIN "zq_zfba_ajxx" zza ON jq."caseno" =zza."ajxx_jqbh" 
            WHERE 1=1 
            AND jq."calltime" BETWEEN '2025-01-01' AND '2025-02-02'
            AND jq.neworicharasubclass IN (
                SELECT unnest(ctc.newcharasubclass_list) 
                FROM ywdata.case_type_config ctc 
                WHERE ctc.leixing = ANY('打架斗殴'))
        ```
