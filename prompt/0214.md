import requests

from tqdm import tqdm  # 进度条库

# 2. 本地大模型 API 地址 (llama.cpp server)
API_URL = "http://127.0.0.1:8080/completion"

# 3. 定义提示词模板 (Prompt)
# 专门针对涉未性侵案件设计，强制输出JSON
PROMPT_TEMPLATE = """

    "接触渠道": "网络游戏",
    "案发场所": "电竞酒店",

    "受害人状态": "离家出走"

def clean_deepseek_response(raw_text):
    """
    清洗 DeepSeek-R1 的输出
    1. 去除 <think>...</think> 标签及其内容
    2. 提取 JSON 字符串
    """
    # 去除思考过程
    cleaned = re.sub(r'<think>.*?</think>', '', raw_text, flags=re.DOTALL).strip()
    
    # 尝试提取 JSON 部分 (找到第一个 { 和最后一个 })
    json_match = re.search(r'\{.*\}', cleaned, re.DOTALL)
    if json_match:
        return json_match.group()
    return cleaned

def query_llm(text):
    """调用本地 llama.cpp 接口"""
    # 截断过长文本，保留前2000字和后3000字，防止爆显存
    # 5060Ti 16G 其实能跑很长，但为了速度，适当截断无伤大雅
    if len(text) > 6000:
        text = text[:2000] + "\n...[中间非关键内容省略]...\n" + text[-3000:]

    prompt = f"<|user|>\n{PROMPT_TEMPLATE.format(context=text)}\n<|assistant|>\n"
    
    payload = {
        "prompt": prompt,
        "n_predict": 512,       # 输出长度限制
        "temperature": 0.1,     # 温度低一点，结果更稳定
        "top_p": 0.9,
        "stop": ["<|end|>", "</s>"] # 停止符
    }
    
    try:
        response = requests.post(API_URL, json=payload, timeout=120)
        if response.status_code == 200:
            result = response.json().get('content', '')
            json_str = clean_deepseek_response(result)
            try:
                return json.loads(json_str)
            except json.JSONDecodeError:
                # 如果解析失败，返回原始文本供人工检查
                return {"解析错误": "JSON格式错误", "原始输出": result[:100]}
        else:
            return {"解析错误": f"API状态码 {response.status_code}"}
    except Exception as e:
        return {"解析错误": str(e)}

# ================= 主程序 =================
if __name__ == "__main__":
    print(f"🚀 开始读取数据: {INPUT_FILE}")
    try:

        df['简要案情'] = ''
    if '笔录内容' not in df.columns:
        print("⚠️ 警告：未找到'笔录内容'列，将使用空字符串代替")
        df['笔录内容'] = ''

    print("   (按 Ctrl+C 可强制停止，已处理的数据会自动保存)")

    # 使用 tqdm 显示进度条
    for index, row in tqdm(df.iterrows(), total=df.shape[0]):
        # 拼接文本：案情 + 笔录
        # 很多时候简要案情里有报警人陈述，也很关键
        context = f"【简要案情】：{str(row['简要案情'])}\n【笔录内容】：{str(row['笔录内容'])}"
        
        # 调用模型
        tags = query_llm(context)
        
        # 将提取的标签合并到当前行
        # 如果模型返回了错误，也会记录下来

        result_row.update(tags)

        # 每10条保存一次，防止意外

    print(f"\n✅ 处理完成！结果已保存至: {OUTPUT_FILE}")
    print(f"📊 共处理 {len(final_df)} 条数据")

def stable_hash(text: str) -> str:
    return hashlib.md5(text.encode("utf-8", errors="ignore")).hexdigest()

def extract_json_str(s: str) -> str:
    s = (s or "").strip()
    if s.startswith("{") and s.endswith("}"):
        return s
    i = s.find("{")
    j = s.rfind("}")
    if i != -1 and j != -1 and j > i:
        return s[i:j+1]
    return s

def safe_json_loads(s: str) -> Optional[Dict[str, Any]]:
    try:
        return json.loads(s)
    except Exception:
        return None

def chunk_text(text: str, chunk_chars: int, overlap_chars: int, max_chunks: int) -> List[Tuple[int, int, str]]:
    text = (text or "").strip()
    if not text:
        return []
    chunks = []
    start = 0
    n = len(text)
    idx = 0
    while start < n and idx < max_chunks:
        end = min(n, start + chunk_chars)
        chunks.append((start, end, text[start:end]))
        idx += 1
        if end >= n:
            break
        start = max(0, end - overlap_chars)
    return chunks

def parse_dt(v: Any) -> Optional[datetime]:
    if v is None:
        return None
    if isinstance(v, datetime):
        return v
    s = str(v).strip()
    if not s:
        return None
    try:
        return dtparser.parse(s, fuzzy=True)
    except Exception:
        return None

def is_valid_cn_id(idno: str) -> bool:
    if not idno:
        return False
    idno = idno.strip()
    return bool(re.fullmatch(r"\d{15}|\d{17}[\dXx]", idno))

def birthdate_from_cn_id(idno: str) -> Optional[date]:
    if not is_valid_cn_id(idno):
        return None
    idno = idno.strip()
    try:
        if len(idno) == 18:
            b = idno[6:14]  # YYYYMMDD
            y, m, d = int(b[0:4]), int(b[4:6]), int(b[6:8])
            return date(y, m, d)
        else:
            b = idno[6:12]  # YYMMDD -> 19YY
            y, m, d = int("19" + b[0:2]), int(b[2:4]), int(b[4:6])
            return date(y, m, d)
    except Exception:
        return None

def calc_age_on(birth: date, on_date: date) -> int:
    years = on_date.year - birth.year
    if (on_date.month, on_date.day) < (birth.month, birth.day):
        years -= 1
    return max(0, years)

def extract_from_chunk(ctext: str, chunk_index: int, cstart: int, cend: int) -> LLMResult:
    system_prompt = FIELD_SPEC.strip()
    user_prompt = f"""
下面是讯问内容的一个分块（chunk）。请严格按规范只输出JSON对象。

chunk_index: {chunk_index}
chunk_char_start: {cstart}
chunk_char_end: {cend}

讯问内容chunk：
{ctext}
""".strip()

    last_err = ""
    for _ in range(MAX_RETRY + 1):
        try:
            raw = call_llama_openai_server(system_prompt, user_prompt)
            js = extract_json_str(raw)
            data = safe_json_loads(js)
            if data is None:
                last_err = "JSON parse failed"
                user_prompt += "\n\n你的上次输出不是严格JSON。请只输出一个JSON对象，不要任何解释。"
                time.sleep(SLEEP_BETWEEN_RETRY)
                continue

            data.setdefault("doc_meta", {})
            data["doc_meta"].update({
                "chunk_index": chunk_index,
                "chunk_char_start": cstart,
                "chunk_char_end": cend,
            })

            # 若模型输出缺字段，补空（保证结构稳定）
            if "fields" not in data or not isinstance(data["fields"], dict):
                data = make_empty_output(chunk_index, cstart, cend)
            else:
                empty = make_empty_output(chunk_index, cstart, cend)
                for k, v in empty["fields"].items():
                    if k not in data["fields"]:
                        data["fields"][k] = v

            return LLMResult(ok=True, data=data, raw=raw)
        except Exception as e:
            last_err = str(e)
            time.sleep(SLEEP_BETWEEN_RETRY)

    return LLMResult(ok=False, data=None, raw="", error=last_err)

FIRST_PRIORITY_FIELDS = {
    "relation", "meet_channel", "invite_channel", "transport", "taken_away",
    "place_type", "overnight", "open_room", "lure_main", "runaway_or_nightout",
}
LAST_PRIORITY_FIELDS = {
    "suspect_knew_minor", "threaten", "threaten_type", "record_video", "spread_or_extort",
    "give_payment", "hotel_register", "evidence_types", "report_delay_range",
    "report_channel", "medical_visit", "drugged"

def merge_chunk_results(chunk_jsons: List[Dict[str, Any]]) -> Dict[str, Any]:
    merged: Dict[str, Any] = {
        "merged_fields": {},
        "conflict_flag": False,
        "evidence_quotes": {},
        "chunk_count": len(chunk_jsons),
    }

    field_names = set()
    for cj in chunk_jsons:
        field_names.update(((cj or {}).get("fields") or {}).keys())

    def add_quote(fname: str, quote: str):
        if not quote:
            return
        merged["evidence_quotes"].setdefault(fname, [])
        if quote not in merged["evidence_quotes"][fname]:
            merged["evidence_quotes"][fname].append(quote)

    def collect_values(fname: str) -> List[str]:
        vals: List[str] = []
        for cj in chunk_jsons:
            fobj = (((cj or {}).get("fields") or {}).get(fname))
            if not isinstance(fobj, dict):
                continue
            if "values" in fobj and isinstance(fobj["values"], list):
                for v in fobj["values"]:
                    if isinstance(v, str) and v.strip() and v.strip() != "不详":
                        vals.append(v.strip())
            if "label" in fobj and isinstance(fobj["label"], str):
                v = fobj["label"].strip()
                if v and v != "不详":
                    vals.append(v)
            q = fobj.get("quote", "")
            if isinstance(q, str) and q.strip():
                add_quote(fname, q.strip())
        # 去重保序
        dedup = []
        for v in vals:
            if v not in dedup:
                dedup.append(v)
        return dedup

    for fname in sorted(field_names):
        dedup_vals = collect_values(fname)
        conflict = len(dedup_vals) > 1
        if conflict:
            merged["conflict_flag"] = True

        first_val = dedup_vals[0] if dedup_vals else "不详"
        last_val = dedup_vals[-1] if dedup_vals else "不详"

        if fname in FIRST_PRIORITY_FIELDS:
            resolved = first_val
            resolve_rule = "first"
        elif fname in LAST_PRIORITY_FIELDS:
            resolved = last_val
            resolve_rule = "last"
        else:
            resolved = last_val if conflict else first_val
            resolve_rule = "last_if_conflict_else_first"

        merged["merged_fields"][fname] = {
            "values": dedup_vals,
            "first": first_val,
            "last": last_val,
            "resolved": resolved,
            "conflict": conflict,
            "resolve_rule": resolve_rule,
        }

    return merged

def find_col_index_by_header(ws, header_name: str) -> Optional[int]:
    for col in range(1, ws.max_column + 1):
        v = ws.cell(row=1, column=col).value
        if isinstance(v, str) and v.strip() == header_name:
            return col
    return None

def ensure_col(ws, header_name: str) -> int:
    col = find_col_index_by_header(ws, header_name)
    if col is not None:
        return col
    new_col = ws.max_column + 1
    ws.cell(row=1, column=new_col).value = header_name
    return new_col

def cell_str(v: Any) -> str:

def cell_int(v: Any) -> int:
    try:
        if v is None or str(v).strip() == "":

def normalize_role(role: str) -> str:
    role = (role or "").strip()
    mapping = {"被害人": "受害人", "受害者": "受害人"}
    return mapping.get(role, role or "不详")

def pick_case_level_value(records_sorted: List[Dict[str, Any]], field_name: str, prefer: str) -> str:
    vals = []
    for rec in records_sorted:
        mf = (rec.get("llm_merged") or {}).get("merged_fields") or {}
        f = mf.get(field_name) or {}
        v = f.get("resolved", "不详")
        if v and v != "不详":
            vals.append(v)

    if prefer == "first":
        return vals[0] if vals else "不详"
    if prefer == "last":
        return vals[-1] if vals else "不详"
    if prefer == "any_yes":
        for v in vals:
            if v == "是":
                return "是"
        for v in vals:
            if v == "否":
                return "否"
        return "不详"
    return vals[-1] if vals else "不详"

def aggregate_case_sheet(rows: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_case: Dict[str, List[Dict[str, Any]]] = {}
    for r in rows:
        cid = (r.get("case_id") or "").strip()
        if cid:
            by_case.setdefault(cid, []).append(r)

    out_rows: List[Dict[str, Any]] = []

    for cid, recs in by_case.items():
        recs_sorted = sorted(recs, key=lambda x: (x.get("times", 0), x.get("start_dt") or datetime.min))
        by_role: Dict[str, List[Dict[str, Any]]] = {}
        for r in recs_sorted:
            by_role.setdefault(normalize_role(r.get("role", "不详")), []).append(r)

        victim = by_role.get("受害人", [])
        suspect = by_role.get("嫌疑人", [])
        reporter = by_role.get("报案人", [])

        def pick_from_roles(field, prefer="first"):
            if victim:
                return pick_case_level_value(victim, field, prefer)
            if suspect:
                return pick_case_level_value(suspect, field, prefer)
            return pick_case_level_value(recs_sorted, field, prefer)

        def any_yes(field):
            return pick_case_level_value(recs_sorted, field, "any_yes")

        row = {
            "案件编号": cid,
            "笔录总数": len(recs_sorted),

            "关系": pick_from_roles("relation", "first"),
            "认识渠道": pick_from_roles("meet_channel", "first"),
            "约见方式": pick_from_roles("invite_channel", "first"),
            "案发地点类型": pick_from_roles("place_type", "first"),
            "案发时间段": pick_from_roles("time_range", "first"),
            "案发日期类型": pick_from_roles("date_type", "first"),
            "诱骗方式(主要)": pick_from_roles("lure_main", "first"),

            "是否留宿": any_yes("overnight"),
            "是否开房": any_yes("open_room"),
            "是否饮酒": any_yes("alcohol"),
            "是否用药": any_yes("drugged"),
            "是否威胁": any_yes("threaten"),
            "是否偷拍视频": any_yes("record_video"),
            "是否传播/勒索": any_yes("spread_or_extort"),

            "是否明知未成年": (pick_case_level_value(suspect, "suspect_knew_minor", "last")
                            if suspect else pick_case_level_value(recs_sorted, "suspect_knew_minor", "last")),

            "是否多次发生": pick_from_roles("repeat", "last"),
            "发生次数区间": pick_from_roles("repeat_count_range", "last"),

            "报案渠道": (pick_case_level_value(reporter, "report_channel", "first")
                       if reporter else pick_case_level_value(recs_sorted, "report_channel", "first")),
            "报案延迟区间": (pick_case_level_value(reporter, "report_delay_range", "first")
                         if reporter else pick_case_level_value(recs_sorted, "report_delay_range", "first")),
            "是否就医": any_yes("medical_visit"),

            "冲突标记": ("是" if any((r.get("llm_merged") or {}).get("conflict_flag") for r in recs_sorted) else "否"),
        }
        out_rows.append(row)

    return out_rows

def process_workbook(path: str):
    wb = load_workbook(path)
    ws = wb[SHEET_NAME] if SHEET_NAME else wb.active

    # 必需列
    col_case = find_col_index_by_header(ws, COL_CASE_ID)
    col_role = find_col_index_by_header(ws, COL_ROLENAME)
    col_times = find_col_index_by_header(ws, COL_TIMES)
    col_start = find_col_index_by_header(ws, COL_START_TIME)

    if None in [col_case, col_role, col_times, col_start]:
        raise ValueError("缺少必要列：案件编号/人员类型/笔录次数/开始时间，请检查列名。")

    # 询问内容列自动识别
    col_text = None
    for name in TRANSCRIPT_CANDIDATES:
        col_text = find_col_index_by_header(ws, name)
        if col_text is not None:
            print(f"使用询问内容列：{name}")
            break
    if col_text is None:
        raise ValueError(f"找不到询问内容列，已尝试：{TRANSCRIPT_CANDIDATES}")

    # 可选列
    col_idno = find_col_index_by_header(ws, COL_IDNO)
    col_age = find_col_index_by_header(ws, COL_AGE)

    # 输出列
    out_all = ensure_col(ws, OUT_COL_JSON_ALL)
    out_merged = ensure_col(ws, OUT_COL_JSON_MERGED)
    out_sum = ensure_col(ws, OUT_COL_SUMMARY)
    out_conf = ensure_col(ws, OUT_COL_CONFLICT)
    out_age_fixed = ensure_col(ws, OUT_COL_AGE_FIXED) if (col_idno and col_age) else None

    cache: Dict[str, Dict[str, Any]] = {}
    rows_for_agg: List[Dict[str, Any]] = []

    total_rows = ws.max_row - 1
    done = 0

    for r in range(2, ws.max_row + 1):
        cid = cell_str(ws.cell(row=r, column=col_case).value).strip()
        role = normalize_role(cell_str(ws.cell(row=r, column=col_role).value))
        times = cell_int(ws.cell(row=r, column=col_times).value)
        start_dt = parse_dt(ws.cell(row=r, column=col_start).value)

        text = cell_str(ws.cell(row=r, column=col_text).value).strip()
        if not text:
            continue

        # 年龄修复：年龄为0/空 且身份证有效 且有开始时间
        if out_age_fixed is not None:
            age_val = cell_int(ws.cell(row=r, column=col_age).value)
            idno = cell_str(ws.cell(row=r, column=col_idno).value).strip()
            if (age_val <= 0) and is_valid_cn_id(idno) and start_dt:
                bd = birthdate_from_cn_id(idno)
                if bd:
                    ws.cell(row=r, column=out_age_fixed).value = calc_age_on(bd, start_dt.date())
                else:
                    ws.cell(row=r, column=out_age_fixed).value = age_val if age_val else ""
            else:
                ws.cell(row=r, column=out_age_fixed).value = age_val if age_val else ""

        # LLM：缓存去重
        h = stable_hash(text)
        if h in cache:
            pack = cache[h]
        else:
            chunks = chunk_text(text, CHUNK_CHARS, OVERLAP_CHARS, MAX_CHUNKS)
            chunk_results = []
            for idx, (cstart, cend, ctext) in enumerate(chunks):
                res = extract_from_chunk(ctext, idx, cstart, cend)
                if not res.ok or not res.data:
                    chunk_results.append(make_empty_output(idx, cstart, cend))
                else:
                    chunk_results.append(res.data)

            merged = merge_chunk_results(chunk_results)

            mf = merged["merged_fields"]
            def rv(k: str) -> str:
                return (mf.get(k, {}) or {}).get("resolved", "不详")

            summary = (
                f"关系:{rv('relation')} 渠道:{rv('meet_channel')} 地点:{rv('place_type')} "
                f"时段:{rv('time_range')} 留宿:{rv('overnight')} 开房:{rv('open_room')} "
                f"诱骗:{rv('lure_main')} 偷拍:{rv('record_video')} 威胁:{rv('threaten')} "
                f"明知未成年:{rv('suspect_knew_minor')} 报案延迟:{rv('report_delay_range')}"
            )

            pack = {"chunk_results": chunk_results, "merged": merged, "summary": summary}
            cache[h] = pack

        # 写回笔录级
        ws.cell(row=r, column=out_all).value = json.dumps(pack["chunk_results"], ensure_ascii=False)
        ws.cell(row=r, column=out_merged).value = json.dumps(pack["merged"], ensure_ascii=False)
        ws.cell(row=r, column=out_sum).value = pack["summary"]
        ws.cell(row=r, column=out_conf).value = "是" if pack["merged"].get("conflict_flag") else "否"

        rows_for_agg.append({
            "case_id": cid,
            "role": role,
            "times": times,
            "start_dt": start_dt,
            "llm_merged": pack["merged"],
        })

        done += 1
        if done % 5 == 0:
            print(f"[{done}/{total_rows}] processed...")

    # 案件级sheet
    case_rows = aggregate_case_sheet(rows_for_agg)
    if "案件级汇总" in wb.sheetnames:
        del wb["案件级汇总"]
    ws_case = wb.create_sheet("案件级汇总")

    if case_rows:
        headers = list(case_rows[0].keys())
        for c, h in enumerate(headers, start=1):
            ws_case.cell(row=1, column=c).value = h
        for i, row in enumerate(case_rows, start=2):
            for c, h in enumerate(headers, start=1):
                ws_case.cell(row=i, column=c).value = row.get(h, "")

    out_path = re.sub(r"\.xlsx$", "", path, flags=re.I) + "_with_llm.xlsx"
    wb.save(out_path)
    print(f"完成，已输出：{out_path}")

构建三类子模型：

统计：

* 跨区域流动次数

引入时间衰减函数：

风险权重 = 原始权重 × e^(-λt)

强化近期行为影响。

#### （3）YOLO行为识别辅助模型

### 2. 图谱关联分析引擎

* 节点：人员

关系权重计算：

权重 = 共现次数 × 时间衰减系数

分析内容包括：

* 二级近邻
* 圈层密度
* 核心度（Degree Centrality）

自动识别：

* 潜在团伙成员
