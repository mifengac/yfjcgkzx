    WITH base_jq AS (
            SELECT
                jq.caseno,
                jq.calltime,
                jq.cmdname,
                jq.dutydeptname,
                jq.occuraddress,
                jq.replies,
                jq.lngoflocate,
                jq.latoflocate,
                jq.casemarkok,
                jq.leixing
            FROM ywdata.v_jq_optimized jq
            ),
            aj AS (
            SELECT *
            FROM ywdata.mv_zfba_all_ajxx
            ),
            ry_agg AS (
            SELECT
                r.asjbh,
                string_agg(
                (COALESCE(r.xm, '') || ',' || '年龄:' || COALESCE(r.nl::text, '') || '(' || COALESCE(r.asjxgry_jsmc, '') || ')'),
                ';'
                ) AS ry_list
            FROM ywdata.zfba_ry_002 r
            GROUP BY r.asjbh
            ),
            ws_agg AS (
            SELECT
                w.asjbh,
                json_agg(
                json_build_object('name', w.flws_zlmc, 'url', w.flwslldz)
                ORDER BY w.wsywxxid
                ) AS ws_json_list,
                count(*) FILTER (WHERE w.flws_zlmc LIKE '%逮捕申请%') AS daibu_cnt,
                count(*) FILTER (WHERE w.flws_zlmc LIKE '%起诉意见书%') AS qisu_cnt,
                count(*) FILTER (WHERE w.flws_dxlxdm = '01' AND w.flws_zlmc LIKE '%起诉意见书%') AS yisong_cnt
            FROM ywdata.zfba_ws_001 w
            WHERE w.flws_lzztdm = '04'
            GROUP BY w.asjbh
            ),
            aj009_agg AS (
            SELECT
                a.asjbh,
                count(*) FILTER (WHERE a.jlts > 0) AS jvliu_cnt,
                count(*) FILTER (WHERE a.fk > 0) AS fakuan_cnt,
                count(*) FILTER (WHERE a.sfjg::NUMERIC > 0) AS jinggao_cnt
            FROM ywdata.zfba_aj_009 a
            GROUP BY a.asjbh
            )
            SELECT
            b.caseno                               AS 警情编号,
            b.leixing                              AS 类型,
            b.calltime                             AS 报警时间,
            b.cmdname                              AS 地区,
            b.dutydeptname                         AS 派出所,
            b.occuraddress                         AS 警情地址,
            b.replies                              AS 处警情况,
            b.lngoflocate                          AS 经度,
            b.latoflocate                          AS 纬度,
            b.casemarkok                           AS 警情标注,
            a."案件编号"                            AS 案件编号,
            a."案件名称"                            AS 案件名称,
            a."简要案情"                            AS 简要案情,
            a."案件类型"                            AS 案件类型,
            a."立案日期"                            AS 立案日期,
            a."案由"                                AS 案由,
            a."案件状态名称"                        AS 案件状态名称,
            r.ry_list                              AS 人员列表,
            w.ws_json_list                         AS 文书列表,
            COALESCE(w.daibu_cnt, 0)               AS 逮捕人数,
            COALESCE(w.qisu_cnt, 0)                AS 起诉人数,
            COALESCE(w.yisong_cnt, 0)              AS 移送人数,
            COALESCE(a9.jvliu_cnt, 0)              AS 拘留人数,
            COALESCE(a9.fakuan_cnt, 0)             AS 罚款人数,
            COALESCE(a9.jinggao_cnt, 0)            AS 警告人数
            FROM base_jq b
            LEFT JOIN aj a
            ON b.caseno = a."警情编号"
            LEFT JOIN ry_agg r
            ON a."案件编号" = r.asjbh
            LEFT JOIN ws_agg w
            ON a."案件编号" = w.asjbh
            LEFT JOIN aj009_agg a9
            ON a."案件编号" = a9.asjbh
    WHERE 1=1

            1. 如果'警情案件类型'有值则在SQL后面加 'AND b.leixing={警情案件类型}'
            2. '报警开始时间','报警结束时间'默认为近3日的0时至今日的0时,如今天是'2025-12-24'则开始时间为'2025-12-21 00:00:00'结束时间为'2025-12-24 00:00:00',SQL条件为'AND b.calltime BETWEEN {开始时间}and {结束时间}'
        2. 对查询到的SQL结果集使用python进行二次统计,统计方式均为计数,逻辑为:
            1. 行标题为对SQL结果集的'地区'列进行分组,最后加一行'合计'
            2. 列标题为固定的:
                1. 警情数,count(警情编号)
                2. 案件数,count(case when 案件编号 is not null then 案件编号)
                3. 案件状态名称:先对SQL结果集中案件状态名称的值进行分组,然后在用案件编号去重计数,动态显示多列.如:分组结果有'已立案','已受案','已调解'三个值,那么后面的列标题依次为'已立案'(count(distinct case when 案件状态名称='已立案' then 案件编号)),'已受案'(count(distinct case when 案件状态名称='已受案' then 案件编号)),'已调解'(count(distinct case when 案件状态名称='已调解' then 案件编号));如分组结果只有'已立案','已受案',那么就显示'已立案','已受案'2列
        3. 如对地区的分组有'云城分局','云安分局','罗定市公安局','新兴县公安局','郁南县公安局',案件状态名称的分组有'已立案','已受案'那么{jqajzl_hz}的显示结果为:

            | 云安分局   |     |     |     |     |
            | 罗定市公安局 |     |     |     |     |
            | 新兴县公安局 |     |     |     |     |
            | 郁南县公安局 |     |     |     |     |
            | **合计** |     |     |     |     |

    1. 判断是否存在'【结警反馈】',若存在则:
        1. 删除'【结警反馈】'之前的内容
        2. 删除'【结警反馈】'之后到下一个空行之后的内容
        3. 例如字段内容为:'XXXXXX【结警反馈】XXXXXXX{空行}',清洗后内容为'【结警反馈】XXXXXXX'
    2. 如果不存在'【结警反馈】'则判断是否存在'【过程反馈】',若存在则:
        1. 删除'【过程反馈】'之前的内容
        2. 删除'【过程反馈】'之后到下一个空行之后的内容
        3. 例如字段内容为:'XXXXXX【过程反馈】XXXXXXX{空行}',清洗后内容为'【过程反馈】XXXXXXX'
    3. 如'【结警反馈】',【过程反馈】'都不存在则判断是否存在'关联重复报警',若存在,则将该字段内容设置为'重复报警'
    4. 如如'【结警反馈】',【过程反馈】','关联重复报警'都不存在则判断是否存在'处理结果说明',若存在
        1. 删除'处理结果说明'之前的内容
        2. 删除'处理结果说明'之后到下一个空行之后的内容
        3. 例如字段内容为:'XXXXXX处理结果说明XXXXXXX{空行}',清洗后内容为'处理结果说明XXXXXXX'
    5. 如'【结警反馈】',【过程反馈】','关联重复报警','处理结果说明'都不存在则设置字段内容为'未反馈'
